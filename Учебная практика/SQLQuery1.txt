CREATE TABLE Филиал (
IDfilial INT IDENTITY (1,1) CONSTRAINT PK1 PRIMARY KEY (IDfilial),
Filial VARCHAR(20),
INNfilial CHAR(10),
Chef VARCHAR(60),
Capacity TINYINT,
Address VARCHAR(60),
Phone CHAR(10)
);

CREATE TABLE Клиент (
ClientID INT IDENTITY (1,1) CONSTRAINT PK2 PRIMARY KEY (ClientID),
Name VARCHAR(60),
LastName VARCHAR(60),
Patronimic VARCHAR(60),
Sign BIT,
Bank VARCHAR(60),
Account VARCHAR(20),
Comment NVARCHAR
);

CREATE TABLE Кузов (
BodyID INT IDENTITY (1,1) CONSTRAINT PK3 PRIMARY KEY (BodyID),
BodyModel VARCHAR(20),
Helm BIT,
Type VARCHAR(15)
);

CREATE TABLE Двигатели (
EngineID INT IDENTITY (1,1) CONSTRAINT PK4 PRIMARY KEY (EngineID),
Volume TINYINT,
Power TINYINT
);

CREATE TABLE Автомобиль (
CarID INT IDENTITY (1,1) CONSTRAINT PK5 PRIMARY KEY (CarID),
Model VARCHAR(15),
BodyID INT CONSTRAINT FK1 FOREIGN KEY (BodyID) REFERENCES Кузов(BodyID),
EngineID INT CONSTRAINT FK2 FOREIGN KEY (EngineID) REFERENCES Двигатели(EngineID),
Drive BIT,
DateStart DATE,
Cost SMALLMONEY,
New BIT,
Year SMALLINT,
Distance TINYINT
);

CREATE TABLE Заказы (
OrderID INT IDENTITY (1,1) CONSTRAINT PK6 PRIMARY KEY (OrderID),
IDfilial INT CONSTRAINT FK3 FOREIGN KEY (IDfilial) REFERENCES Филиал(IDfilial),
ClientID INT CONSTRAINT FK4 FOREIGN KEY (ClientID) REFERENCES Клиент(ClientID),
CarID INT CONSTRAINT FK5 FOREIGN KEY (CarID) REFERENCES Автомобиль(CarID),
StartDate DATE,
Price SMALLMONEY
);
